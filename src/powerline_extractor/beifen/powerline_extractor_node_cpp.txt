#include "powerline_extractor.h"

int main(int argc, char** argv) {
    ros::init(argc, argv, "powerline_extractor_node");
    ros::NodeHandle nh;
    
    PowerlineExtractor extractor;
    
    // 设置1Hz的发布频率
    ros::Rate rate(1); // 1 Hz
    while (ros::ok()) {
        extractor.processAndPublishPowerlines();
        ros::spinOnce();
        rate.sleep();
    }
    
    return 0;
}
